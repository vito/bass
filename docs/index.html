(def concourse
  (load (.concourse)))

(def booklit
  (concourse:resource :git {:uri "https://github.com/vito/booklit"}))

(let [latest-booklit (concourse:get-latest booklit)]
  (-> ($ latest-booklit/scripts/build-docs $latest-booklit)
      (in-image "golang")
      (response-from :exit)
      (path ./docs/)
      (emit *stdout*)))
